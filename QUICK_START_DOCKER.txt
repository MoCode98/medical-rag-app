┌──────────────────────────────────────────────────────────────────────────────┐
│ 1. ENSURE DOCKER IS INSTALLED                                               │
└──────────────────────────────────────────────────────────────────────────────┘

  # Check Docker is running
  docker --version
  docker compose version

  # If not installed:
  # Windows/Mac: Download Docker Desktop from https://docker.com
  # Linux: sudo apt-get install docker-compose-plugin

┌──────────────────────────────────────────────────────────────────────────────┐
│ 2. (OPTIONAL) ADD PDFs FOR AUTO-INGESTION                                   │
└──────────────────────────────────────────────────────────────────────────────┘

  # Drop PDF files into the pdfs/ folder (created automatically)
  # They will be auto-ingested on container startup!

┌──────────────────────────────────────────────────────────────────────────────┐
│ 3. START THE CONTAINERS                                                     │
└──────────────────────────────────────────────────────────────────────────────┘

  docker compose up -d

  # First run downloads:
  # - Ollama container (~2GB)
  # - Builds RAG app (~500MB)
  # Takes 10-15 minutes on first run
  # Subsequent runs: < 30 seconds

┌──────────────────────────────────────────────────────────────────────────────┐
│ 4. DOWNLOAD AI MODELS (ONE-TIME SETUP)                                      │
└──────────────────────────────────────────────────────────────────────────────┘

  # Embedding model (274MB, 2-5 minutes)
  docker compose exec ollama ollama pull nomic-embed-text

  # LLM model (4.7GB, 10-30 minutes)
  docker compose exec ollama ollama pull deepseek-r1:7b

  # Verify models are downloaded
  docker compose exec ollama ollama list

┌──────────────────────────────────────────────────────────────────────────────┐
│ 5. OPEN IN BROWSER                                                          │
└──────────────────────────────────────────────────────────────────────────────┘

  http://localhost:8000

┌──────────────────────────────────────────────────────────────────────────────┐
│ KEY FEATURES                                                                 │
└──────────────────────────────────────────────────────────────────────────────┘

  🐳 DOCKER BENEFITS
     • Complete isolation - no conflicts with system packages
     • Cross-platform - same setup on Windows, Mac, Linux
     • Easy cleanup - remove everything with one command
     • Persistent data - volumes keep your data safe
     • Auto-restart - containers restart after reboot (unless stopped)

  📁 VOLUME MOUNTS
     • ./pdfs/ → Container's pdfs/ folder (auto-ingestion)
     • ./logs/ → Application logs on host machine
     • vector_db → Named volume (critical - contains embeddings)
     • data_cache → Named volume (caches, performance boost)
     • ollama_data → Named volume (AI models, 5GB+)

  🚀 AUTO-INGESTION
     • Drop PDFs in ./pdfs/ folder
     • Restart container: docker compose restart rag-app
     • Ingestion runs in background
     • UI immediately accessible

┌──────────────────────────────────────────────────────────────────────────────┐
│ COMMON COMMANDS                                                               │
└──────────────────────────────────────────────────────────────────────────────┘

  # View running containers
  docker compose ps

  # View logs (all services)
  docker compose logs

  # Follow logs (real-time)
  docker compose logs -f rag-app

  # Stop containers (preserves data)
  docker compose stop

  # Stop and remove containers (keeps volumes/data)
  docker compose down

  # Restart a service
  docker compose restart rag-app

  # Rebuild after code changes
  docker compose build --no-cache
  docker compose up -d

  # Execute command in container
  docker compose exec rag-app python query.py --help

┌──────────────────────────────────────────────────────────────────────────────┐
│ TROUBLESHOOTING                                                               │
└──────────────────────────────────────────────────────────────────────────────┘

  ❌ "Cannot connect to Docker daemon"
     → Ensure Docker Desktop is running
     → Check system tray for Docker icon

  ❌ "Port 8000 already in use"
     → docker compose down
     → Or edit docker-compose.yml to use different port (8080:8000)

  ❌ "Cannot connect to Ollama"
     → docker compose restart ollama
     → Ensure models are downloaded (see step 4)
     → Check logs: docker compose logs ollama

  ❌ "Vector database is empty"
     → Add PDFs to ./pdfs/ folder
     → Restart: docker compose restart rag-app
     → Or use Ingest tab to upload manually

  ❌ Container shows "unhealthy"
     → docker compose logs rag-app
     → docker compose restart
     → If persists: docker compose down && docker compose up -d

  ❌ Slow performance
     → Increase Docker resources:
       Docker Desktop → Settings → Resources → Advanced
       Set: 4+ CPUs, 8GB+ RAM
     → Use smaller model: ollama pull qwen:0.5b

┌──────────────────────────────────────────────────────────────────────────────┐
│ DATA MANAGEMENT                                                               │
└──────────────────────────────────────────────────────────────────────────────┘

  # Backup vector database (most important)
  docker compose cp rag-app:/app/vector_db ./backup/vector_db

  # Backup data cache
  docker compose cp rag-app:/app/data ./backup/data

  # Restore from backup
  docker compose cp ./backup/vector_db rag-app:/app/vector_db
  docker compose cp ./backup/data rag-app:/app/data

  # Complete cleanup (⚠️ DELETES ALL DATA INCLUDING VOLUMES)
  docker compose down -v

  # View disk space used
  docker system df

  # Clean up unused resources
  docker system prune

┌──────────────────────────────────────────────────────────────────────────────┐
│ CUSTOMIZATION                                                                 │
└──────────────────────────────────────────────────────────────────────────────┘

  # Edit docker-compose.yml to customize:

  environment:
    - OLLAMA_MODEL=qwen:7b              # Change AI model
    - OLLAMA_EMBEDDING_MODEL=nomic-embed-text
    - CHUNK_SIZE=512                     # Adjust chunking
    - TOP_K_RESULTS=5                    # Number of chunks
    - TEMPERATURE=0.7                    # AI creativity

  # After editing, restart:
  docker compose down
  docker compose up -d
